Example

________________________________________________________
Answer: Run length encoding
a) Pick the first character from the string. 
b) Append the picked character to the destination string. 
c) Count the number of occurrences of the picked character and append the count to the destination string.
without recursion:

function compString(str)
  {
    let n = str.length;
    for ( let i = 0; i < n; i++)
    {
      //this will count the number of times the char is repeated
      let count = 1;
    while (i < n - 1 && str[i] == str[i+1])
      {
        count++;
        i++;
      }

    }

Answer4: Checking for Uniqueness

function uniqChar(str)
  {
    for(let i = 0; i < str.length; i++) //run 2 loops
      for(j = i + 1; j < str.length; j++)
        if (str[i] == str[j]) //if they become = at any point return false
            return false;


    return true;
  }
